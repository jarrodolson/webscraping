CREATE TABLE cache (
	id serial PRIMARY KEY,
	createdate timestamp DEFAULT current_timestamp,
	url text NOT NULL UNIQUE,
	protocol text,
	domain text,
	status text,
	status_msg text,
	download_date timestamp,
	user_agent text,
	url_returned text,
	protocol_returned text,
	domain_returned text,
	server_returned text,
	date_on_server timestamp,
	content_type text,
	transfer_encoding text,
	connection text,
	vary text,
	x_frame_options text,
	set_cookie text,
	full_txt text,
	full_txt_hash text UNIQUE,
	head text,
	body text,
	body_clean text
);
